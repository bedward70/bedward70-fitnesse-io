!define encoding {UTF-8}

|!-B70DoFixture-!|

|saveSymbol|read|World|$second-world|


|!-B70JoinFixture-!|=$first-world|!-, -!|=$second-world|

|saveSymbol|join|$phrase|


|!-B70DoFixture-!|
|check|readSymbol|$phrase|Hello, World!|

Save to File

|!-B70FileFixture-!|
|set encoding|${encoding}|

|!-B70FileFixture-!|
|set encoding|${encoding}|

|!-B70DoFixture-!|
|check|readSymbol|$saved-phrase|Hello, World!|

|!-B70FileFixture-!|
